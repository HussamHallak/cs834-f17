function gazetteerLookup(term)
	placeNames = []
	//fill the array placeNames with placeNames
	placeNames.append('Norfolk')
	placeNames.append('Suffolk')
	placeNames.append('Chesapeake')
	.
	.
	.
	if term in placeNames:
		return True
	else:
		return False
	
function getPlaceNames(query):
	result = []
	tokens = query.split()
	preps = ['in', 'from', 'near', 'to', 'of']
	for i in range(0, len(tokens)):
		for prep in preps:
			if prep = tokens[i]:
				if gazetteerLookup(tokens[i+1]):
					result.append(tokens[i+1])
	return result
